tages:
  - build
  - deploy

build pre_prod:
  image: node:14-slim
  stage: build
  only:
    - develop
    - master
  script:
    - npm install
    - npm run build
    - mv build dist
    - echo "BUILD SUCCESSFULLY"
  artifacts:
    paths:
      - dist/
    expire_in: 1 hour
  
deploy pre_prod:
  image: python:latest
  stage: deploy
  script:
    - export AWS_ACCESS_KEY_ID=$AWS_ACCESS_KEY_ID 
    - export AWS_SECRET_ACCESS_KEY=$AWS_SECRET_ACCESS_KEY 
    - pip install awscli
    - aws s3 sync ./dist s3://sp-preprod-api-website
    - aws s3 sync --cache-control 'max-age=604800' --exclude index.html dist/ s3://sp-preprod-api-website/
    - aws s3 sync --cache-control 'no-cache' dist/ s3://sp-preprod-api-website/
    - echo "DEPLOYED SUCCESSFULLY"
  environment:
    name: preprod
  only:
    - develop
